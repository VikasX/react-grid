Custom column formatters¶

var FancyCell = React.createClass({
  render: function() {
    return (
      <div style={{fontWeight: 'bold', color: 'red'}}>
        № {this.props.value}
      </div>
    );
  }
});

var columns = [
  {
    name: '№',
    width: '20%',
    key: 0,
    formatter: FancyCell
  },
  {
    name: 'Name',
    width: '30%',
    key: 1
  },
  {
    name: 'Surname',
    width: '50%',
    key: 2
  }
];
